686 i++;
687 if (i)
688 {

690 last.getNode()->prev = splice_begin->prev;
691 (position.getNode()->prev)->next = splice_begin;
692 splice_begin->prev = position.getNode()->prev;
693 (position.getNode())->prev = splice_end;

695 x._size -= i;

853 void sort(Compare comp)
854 {
855 if (this->size() <= 1)
856 return ;
857 for (iterator it = this->begin(); it != this->end(); it++)
858 {
859 for (iterator itr = it; itr != this->end(); itr++)

911 void swap_order(Node* A, Node* B)
912 {
913 Node *ptrVec[] = {A->prev, B->prev, A->next, B->next};
914
915 if (A == B)
916 return;
917 if (A->next == B || B->next == A)
918 {
919 A->prev = ptrVec[2];
920 B->prev = ptrVec[0];
921 A->next = ptrVec[3];
922 B->next = ptrVec[1];
923 }
924 else
925 {
926 A->prev = ptrVec[1];
927 B->prev = ptrVec[0];
928 A->next = ptrVec[3];
929 B->next = ptrVec[2];
930 }
931 A->prev->next = A->next->prev = A;
932 B->prev->next = B->next->prev = B;
933 }
