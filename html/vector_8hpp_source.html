587 void insert(iterator position, size_type n, const value_type& val)
588 {
589 if (n == 0)
590 return;
591 difference_type shift = position - this->begin();
592 difference_type tmp = this->end() - this->begin();
593
594 this->resize(this->_size + n);
595 iterator end = this->end();
596 position = this->begin() + shift;
597 iterator tmp_end = this->begin() + tmp;
598 while (tmp_end != position)
599 {
600 --end;
601 --tmp_end;
602 *end = *tmp_end;
603 }
604 while (n--)
605 *position++ = val;
606 }

619 void insert(iterator position, typename ft::enable_if
620 <!std::numeric_limits<InputIterator>::is_integer, InputIterator>
621 ::type first, InputIterator last)
622 {
623 difference_type shift = (position - this->begin());
624 size_type n = 0;
625 InputIterator tmp(first);
626 while (tmp != last)
627 {
628 ++tmp;
629 ++n;
630 }
631 if ((_size + n) > _capacity)
632 reallocate(_size + n);
633 _size += n;
634 iterator it(&_head[_size - n]);

734 size_type grow(size_type new_size)
735 {
736 size_type growth_factor = 2;
737
738 size_type new_capacity = (_capacity > 0 ? _capacity : 1);
739 while (new_capacity < new_size)
740 new_capacity *= growth_factor;
741 return (new_capacity);
742 }

747 void reallocate(size_type new_capacity)
748 {
749 pointer new_vector = _alloc.allocate(new_capacity);
750 for (size_type i = 0; i < _size; i++)
751 {
752 _alloc.construct(&new_vector[i], _head[i]);
753 _alloc.destroy(&_head[i]);
754 }
755 if (_capacity)
756 _alloc.deallocate(_head, _capacity);
757 _head = new_vector;
758 _capacity = new_capacity;
759 }
